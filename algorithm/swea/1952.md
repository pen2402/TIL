# [모의 SW 역량테스트] 수영장

```python
def comb(idx, p):
    global res
    if idx >= 12:
        res = min(res, p)  # 더 적은 비용이면 갱신
        return
    if p > res:  # 이전 비용보다 더 높으면 함수 종료
        return

    comb(idx+1, p+ min(plan[idx]*price[0], price[1]))
    comb(idx+3, p+price[2])

T = int(input())
for t in range(1, T+1):
    price = list(map(int, input().split()))
    plan = list(map(int, input().split()))
    res = price[3]
    comb(0, 0)
    print(f'#{t} {res}')
```
