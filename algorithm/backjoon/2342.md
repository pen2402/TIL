# Dance Dance Revolution

```python
power = [
    (0, 2, 2, 2, 2),
    (0, 1, 3, 4, 3),
    (0, 3, 1, 3, 4),
    (0, 4, 3, 1, 3),
    (0, 3, 4, 3, 1),
]

def solution():
    numbers = tuple(map(int, input().split()))
    N = len(numbers)
    dp = [[[400000]*5 for _ in range(5)] for _ in range(N+1)]
    dp[0][0][0] = 0
    for i in range(N):
        for j in range(5):
            for k in range(5):
                if dp[i][j][k]+power[j][numbers[i]] < dp[i+1][numbers[i]][k]:
                    dp[i+1][numbers[i]][k] = dp[i][j][k]+power[j][numbers[i]]
                if dp[i][j][k]+power[k][numbers[i]] < dp[i+1][j][numbers[i]]:
                    dp[i+1][j][numbers[i]] = dp[i][j][k]+power[k][numbers[i]]
    print(min(dp[-1][0]))

solution()
```
