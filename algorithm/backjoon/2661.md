# 좋은수열

```python
def dfs(l, s):
    global res
    if res:
        return
    for i in range(l//2):
        if s[l-i-1:] == s[l-(i+1)*2:l-i-1]:
            return
    if l >= N:
        res = s
        return
    dfs(l+1, s+'1')
    dfs(l+1, s+'2')
    dfs(l+1, s+'3')

N = int(input())
res = ''
dfs(1, '1')
print(res)
```

    if (l >= N) {
        res = s;
        return;
    }
    dfs(l+1, s+'1')
    dfs(l+1, s+'2')
