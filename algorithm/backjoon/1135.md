# 뉴스 전하기

```python
def dfs(idx):
    temp = []
    for i in child[idx]:
        dfs(i)
        temp.append(dp[i])
    if not temp:
        temp.append(0)
    temp.sort(reverse=True)
    time = [temp[i]+i+1 for i in range(len(temp))]
    dp[idx] = max(time)

N = int(input())
child = [[] for _ in range(N)]
P = list(map(int, input().split()))
for i in range(1, N):
    child[P[i]].append(i)
dp = [0]*N
dfs(0)
print(dp[0]-1)
```
