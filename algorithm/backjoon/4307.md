```python
import sys

input = sys.stdin.readline

def solution():
    T = int(input())
    for _ in range(T):
        l, n = map(int, input().split())
        min_val = max_val = 0
        for _ in range(n):
            ant = int(input())
            a = ant
            b = l-ant
            if a > b:
                a, b = b, a
            if a > min_val:
                min_val = a
            if b > max_val:
                max_val = b
        print(min_val, max_val)

solution()
```
