# 세 수

```python
import sys

input = sys.stdin.readline

def solution():
    a, b, c = map(int, input().split())
    if a+b == c:
        print(f'{a}+{b}={c}')
    elif a-b == c:
        print(f'{a}-{b}={c}')
    elif a*b == c:
        print(f'{a}*{b}={c}')
    elif a//b == c and a%b == 0:
        print(f'{a}/{b}={c}')
    elif a == b+c:
        print(f'{a}={b}+{c}')
    elif a == b-c:
        print(f'{a}={b}-{c}')
    elif a == b*c:
        print(f'{a}={b}*{c}')
    else:
        print(f'{a}={b}/{c}')

solution()
```
