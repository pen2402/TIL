# 고급 수학

```python
import sys

input = sys.stdin.readline

def solution():
    n = int(input())
    for i in range(1, n+1):
        a, b, c = sorted(map(int, input().split()))
        print(f'Scenario #{i}:')
        print('yes' if a*a+b*b == c*c else 'no')
        if i < n:
            print()

solution()
```
