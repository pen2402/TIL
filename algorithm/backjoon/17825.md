def move(idx, score):
    global res
    if idx > 9:
        if score > res:
            res = score
        return
    visited = set()
    for i in range(4):
        p = pieces[i]
        if p != -1 and p not in visited:
            visited.add(p)
            next_p = board[p][numbers[idx]]
            if next_p == -1 or next_p not in pieces:
                pieces[i] = next_p
                move(idx+1, score+board[next_p][0])
                pieces[i] = p
