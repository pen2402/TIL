# 삼각형과 세 변

```python
import sys

input = sys.stdin.readline

def solution():
    while 1:
        a, b, c = map(int, input().split())
        length = len({a, b, c})
        if a == 0:
            return
        if max(a, b, c)*2 >= a+b+c:
            print('Invalid')
        elif length == 1:
            print('Equilateral')
        elif length == 2:
            print('Isosceles')
        else:
            print('Scalene')

solution()
```
