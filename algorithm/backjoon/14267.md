import sys

input = sys.stdin.readline

def solution():
    n, m = map(int, input().split())
    numbers = tuple(map(int, input().split()))
    dp = [0]*n
    for _ in range(m):
        i, m = map(int, input().split())
        dp[i-1] += m
    for i in range(n):
        if numbers[i] > 0:
            dp[i] += dp[numbers[i]-1]
    print(' '.join(map(str, dp)))

solution()
