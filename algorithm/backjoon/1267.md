# 핸드폰 요금

```python
import sys

input = sys.stdin.readline

def solution():
    N = int(input())
    times = tuple(map(int, input().split()))
    Y = M = 0
    for time in times:
        Y += 10*(time//30+(time%30 >= 0))
        M += 15*(time//60+(time%60 >= 0))
    print(('Y '*(Y <= M)+'M '*(M <= Y)+str(min(Y, M))))

solution()
```
