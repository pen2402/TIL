```python
import sys

input = sys.stdin.readline

def solution():
    resistances = {
        'black': '0', 'brown': '1', 'red': '2', 'orange': '3', 'yellow': '4',
        'green': '5', 'blue': '6', 'violet': '7', 'grey': '8', 'white': '9'
    }
    res = resistances.get(input().rstrip(), '')
    res += resistances.get(input().rstrip(), '')
    res += '0'*int(resistances.get(input().rstrip(), '0'))
    print(int(res))

solution()
```
