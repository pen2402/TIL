# 출석 이벤트

```python
import sys

input = sys.stdin.readline

def solution():
    N = int(input())
    P = int(input())
    res = P
    if N >= 5:
        res -= 500
    if N >= 10 and P//10*9 < res:
        res = P//10*9
    if N >= 15 and P-2000 < res:
        res = P-2000
    if N >= 20 and P//4*3 < res:
        res = P//4*3
    if res < 0:
        res = 0
    print(res)

solution()
```
