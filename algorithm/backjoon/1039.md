def dfs(idx, cnt):
    global res
    if cnt >= K:
        _num = int(''.join(num))
        if _num > res:
            res = _num
        return
    if idx >= M-2:
        if (K-cnt)%2:
            num[-1], num[-2] = num[-2], num[-1]
            dfs(idx, K)
            num[-1], num[-2] = num[-2], num[-1]
        else:
            dfs(idx, K)
        return
    dfs(idx+1, cnt)
    for j in range(idx+1, M):
        num[idx], num[j] = num[j], num[idx]
        dfs(idx+1, cnt+1)
        num[idx], num[j] = num[j], num[idx]

N, K = input().split()
K = int(K)
M = len(N)
res = -1
num = [*N]
if M == 1 or (M == 2 and num[-1] == '0'):
    print(-1)
else:
    dfs(0, 0)
    print(res)
