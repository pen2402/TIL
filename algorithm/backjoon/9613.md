```python
T = int(input())
for t in range(T):
    gcd_list = []
    li = list(map(int, input().split()))
    for i in range(1, li[0]+1):
        j = 2
        gcd = []
        while li[i] > 1:
            if li[i]%j == 0:
                gcd.append(j)
                li[i] = li[i]//j
            else:
                j += 1
        gcd_list.append(gcd)
    
    result = []
    for i in range(len(gcd_list)):
        for j in range(i+1, len(gcd_list)):
            li2 = gcd_list[j].copy()
            res = 1
            for g in gcd_list[i]:
                if g in li2:
                    li2.remove(g)
                    res *= g
            result.append(res)
    print(sum(result))
```
