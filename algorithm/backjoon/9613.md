```python
import sys

input = sys.stdin.readline

def solution():

    def gcd(a, b):
        while a > 0:
            b, a = a, b%a
        return b

    T = int(input())
    for _ in range(T):
        n, *numbers = map(int, input().split())
        print(sum(gcd(numbers[i], numbers[j]) for i in range(n-1) for j in range(i+1, n)))

solution()
```
