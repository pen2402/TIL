```python
import sys

input = sys.stdin.readline

N = int(input())
queue = [0] * 2000000
front = rear = 0
for _ in range(N):
    cmd = list(input().split())
    if cmd[0] == 'push':
        queue[rear] = int(cmd[1])
        rear += 1
    elif cmd[0] == 'pop':
        if rear-front:
            print(queue[front])
            front += 1
        else:
            print(-1)
    elif cmd[0] == 'size':
        print(rear-front)
    elif cmd[0] == 'empty':
        print(0) if rear-front else print(1)
    elif cmd[0] == 'front':
        print(queue[front]) if rear-front else print(-1)
    elif cmd[0] == 'back':
        print(queue[rear-1]) if rear-front else print(-1)
```
