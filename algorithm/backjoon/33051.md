# 나는 이 우마를 지배할 수 있다

```python
import sys

input = sys.stdin.readline

def solution():

    def dfs(idx, total, d_list):
        nonlocal res
        if res:
            return
        if idx >= 3:
            if 100 >= total >= -100 and -total <= d_list[-1]:
                d_list.append(-total)
                p_list = [[-scores[p], p+1] for p in range(4)]
                for i in range(4):
                    for p in range(4):
                        p_list[p][0] -= d_list[i+1]*cnt_list[p][i]
                p_list.sort()
                if p_list[M-1][1] == K:
                    res = d_list
            return
        for d in range(d_list[-1], -101, -1):
            dfs(idx+1, total+d, d_list+[d])

    N, K, M = map(int, input().split())
    cnt_list = [[0]*4 for _ in range(4)]
    scores = [0]*4
    for _ in range(N):
        a1, a2, a3, a4, s1, s2, s3, s4 = map(int, input().split())
        scores[a1-1] += s1
        scores[a2-1] += s2
        scores[a3-1] += s3
        scores[a4-1] += s4
        cnt_list[a1-1][0] += 1
        cnt_list[a2-1][1] += 1
        cnt_list[a3-1][2] += 1
        cnt_list[a4-1][3] += 1
    res = None
    dfs(0, 0, [100])
    if res:
        print(*res[1:])
        return
    print(-1)

solution()
```
