# 적어도 대부분의 배수

input = sys.stdin.readline

def solution():

    def get_gcd(a, b):
        while a > 0:
            r = b%a
            b, a = a, r
        return b

    numbers = tuple(map(int, input().split()))
    res = -1
    for i in range(5):
        for j in range(i+1, 5):
            gcd = get_gcd(numbers[i], numbers[j])
            _num = numbers[i]*numbers[j]//gcd
            for k in range(j+1, 5):
                gcd = get_gcd(_num, numbers[k])
                num = _num*numbers[k]//gcd
                if res < 0 or num < res:
                    res = num
    print(res)

solution()
