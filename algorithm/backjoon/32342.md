# 와우와 쿼리

```python
import sys

input = sys.stdin.readline

def solution():
    Q = int(input())
    for _ in range(Q):
        S = input()
        res = 0
        idx = S.find('WOW')
        while idx >= 0:
            res += 1
            idx = S.find('WOW', idx+2)
        print(res)

solution()
```
