# 정수 a를 k로 만들기

```python
import sys

input = sys.stdin.readline

A, K = map(int, input().split())
visited = [0]*1000001
queue = [A]
visited[A] = 1
while queue:
    n = queue.pop(0)
    if n == K:
        break
    if n+1 <= K and not visited[n+1]:
        visited[n+1] = visited[n]+1
        queue.append(n+1)
    if n*2 <= K and not visited[n*2]:
        visited[n*2] = visited[n]+1
        queue.append(n*2)
print(visited[K]-1)
```
