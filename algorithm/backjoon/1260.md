```python
import sys

N, M, V = map(int, sys.stdin.readline().split())
edge = [list(map(int, sys.stdin.readline().split())) for _ in range(M)]
graph = [[0] * N for _ in range(N)]
for e in edge:
    graph[e[0]-1][e[1]-1] = 1
    graph[e[1]-1][e[0]-1] = 1

visited = [0] * N
visited[V-1] = 1
stack = []
path = [V]
i = V-1
while 1:
    for j in range(N):
        if graph[i][j] and not visited[j]:
            stack.append(i)
            path.append(j+1)
            visited[j] = 1
            i = j
            break
    else:
        if stack:
            i = stack.pop()
        else:
            break
print(*path)

visited = [0] * N
visited[V-1] = 1
queue = [V-1]
path = [V]
while queue:
    i = queue.pop(0)
    for j in range(N):
        if graph[i][j] and not visited[j]:
            queue.append(j)
            path.append(j+1)
            visited[j] = 1
print(*path)
```
