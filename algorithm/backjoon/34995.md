# 쿠폰 잘 사용하기?

```python
import sys

input = sys.stdin.readline

def solution():
    N, S = input().split()
    A = input().rstrip()
    S = S.replace('?', '9')
    N = int(N)
    if len(A) < N:
        print(S)
    elif len(A) > N:
        print(-1)
    else:
        print(S if S >= A else -1)

solution()
```
