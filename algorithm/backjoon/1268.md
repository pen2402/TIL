# 임시 반장 정하기

```python
import sys

input = sys.stdin.readline

def solution():
    N = int(input())
    students = [tuple(map(int, input().split())) for _ in range(N)]
    friends = [[False]*N for _ in range(N)]
    for k in range(5):
        for i in range(N):
            for j in range(i+1, N):
                if students[i][k] == students[j][k]:
                    friends[i][j] = True
                    friends[j][i] = True
    counts = []
    for friend in friends:
        counts.append(sum(friend))
    print(counts.index(max(counts))+1)

solution()
```
