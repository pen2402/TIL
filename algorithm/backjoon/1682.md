# 돌리기

```python
import sys

input = sys.stdin.readline

def solution():
    state = input()[::2]
    if state == '12345678':
        print(0)
        return
    stack = ['12345678']
    visited = {}
    while stack:
        _stack = []
        while stack:
            s = stack.pop()
            for _s in [s[::-1], s[3]+s[:3]+s[5:8]+s[4], s[0]+s[2]+s[5]+s[3]+s[4]+s[6]+s[1]+s[7], s[4]+s[1:4]+s[0]+s[5:8]]:
                if _s not in visited:
                    visited[_s] = visited.get(s, 0)+1
                    if _s == state:
                        print(visited.get(_s, 0))
                        return
                    _stack.append(_s)
        stack = _stack

solution()
```
