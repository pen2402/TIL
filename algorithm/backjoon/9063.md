# 대지

```python
import sys

input = sys.stdin.readline

def solution():
    N = int(input())
    max_x = max_y = -10000
    min_x = min_y = 10000
    for _ in range(N):
        x, y = map(int, input().split())
        if x < min_x:
            min_x = x
        if y < min_y:
            min_y = y
        if x > max_x:
            max_x = x
        if y > max_y:
            max_y = y
    print((max_x-min_x)*(max_y-min_y))

solution()
```
