# 이진수

```python
import sys

input = sys.stdin.readline

def solution():
    T = int(input())
    for _ in range(T):
        n = int(input())
        res = []
        i = 0
        while n > 0:
            if (1 << i)&n:
                res.append(i)
                n -= 1 << i
            i += 1
        print(*res)

solution()
```
