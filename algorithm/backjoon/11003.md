# 최솟값 찾기

```python
import sys
from collections import deque

input = sys.stdin.readline

def solution():
    N, L = map(int, input().split())
    A = tuple(map(int, input().split()))
    queue = deque()
    res = []
    for i in range(N):
        while queue and queue[-1] > A[i]:
            queue.pop()
        queue.append(A[i])
        if i >= L and queue[0] == A[i-L]:
            queue.popleft()
        res.append(f'{queue[0]}')
    print(' '.join(res))

solution()
```
