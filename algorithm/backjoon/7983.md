# 내일 할거야

```python
import sys

input = sys.stdin.readline

def solution():
    n = int(input())
    tasks = []
    for _ in range(n):
        d, t = map(int, input().split())
        tasks.append((d, t))
    tasks.sort(key=lambda x: -x[1])
    res = tasks[0][1]
    for d, t in tasks:
        if t < res:
            res = t
        res -= d
    print(res)

solution()
```
