# 부분수열의 합

```python
def subset(i, total, N):
    global total_list
    if i >= N:
        total_list[total] = 1
        return
    subset(i+1, total+S[i], N)
    subset(i+1, total, N)

N = int(input())
S = list(map(int, input().split()))
res = sum(S)+1
total_list = [0] * res
subset(0, 0, N)
for i in range(1, res):
    if not total_list[i]:
        res = i
        break
print(res)
```
