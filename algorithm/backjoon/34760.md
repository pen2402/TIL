# 코끼리 15마리가 앞으로 돌진하면 어떻게 막을 건데

```python
import sys

input = sys.stdin.readline

def solution():
    res = 0
    flag = False
    for a in map(int, input().split()):
        if a > res:
            res = a
            flag = True
        elif flag:
            res += 1
            flag = False
    print(res)

solution()
```
