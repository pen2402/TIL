def sequence(N, M):
    if M == 0:
        print(*stack)
        return
    for i in range(N):
        if not visited[i]:
            stack.append(A[i])
            visited[i] = 1
            sequence(N, M-1)
            stack.pop()
            visited[i] = 0

N, M = map(int, input().split())
A = list(map(int, input().split()))
A.sort()
visited = [0] * N
stack = []
sequence(N, M)
