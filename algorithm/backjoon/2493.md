```python
import sys

input = sys.stdin.readline

N = int(input())
height = list(map(int, input().split()))
res = [0]*N
stack = []
for i in range(N):
    h = height[i]
    while stack and stack[-1][0] < h:
        stack.pop()
    if stack:
        res[i] = stack[-1][1]+1
    stack.append([h, i])
print(*res)
```
