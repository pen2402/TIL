import sys

input = sys.stdin.readline

def solution():
    T = int(input())
    for _ in range(T):
        S = input().rstrip()
        cnt_list = [0]*26
        for c in S:
            if c != ' ':
                cnt_list[ord(c)-97] += 1
        max_cnt = max(cnt_list)
        print(chr(cnt_list.index(max_cnt)+97) if cnt_list.count(max_cnt) == 1 else '?')

solution()
