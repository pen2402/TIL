def stars(n):
    global star
    if n >= N:
        print('\n'.join(star))
        return
    for i in range(len(star)):
        star[i] = '* ' + star[i] + ' *'
    star = ['*'*(n*4+1), '*' + ' '*(n*4-1) + '*'] + star
    star = star + ['*' + ' '*(n*4-1) + '*', '*'*(n*4+1)]
    return stars(n+1)

N = int(input())
star = ['*']
stars(1)
