# 싸이버개강총회

```python
import sys

S, E, Q = input().split()
S = tuple(map(int, S.split(':')))
E = tuple(map(int, E.split(':')))
Q = tuple(map(int, Q.split(':')))
enter = set()
leave = set()
for case in sys.stdin.readlines():
    T, name = case.split()
    T = tuple(map(int, T.split(':')))
    if T[0] < S[0] or (T[0] == S[0] and T[1] <= S[1]):
        enter.add(name)
    elif T[0] > E[0] or (T[0] == E[0] and T[1] >= E[1]):
        if T[0] < Q[0] or (T[0] == Q[0] and T[1] <= Q[1]):
            if name in enter:
                leave.add(name)
print(len(leave))
```

```javascript
const fs = require('fs');
const input = fs.readFileSync('/dev/stdin').toString().split('\n');

let [S, E, Q] = input[0].split(' ');
S = S.split(':').map(Number);
E = E.split(':').map(Number);
Q = Q.split(':').map(Number);
let enter = new Set();
let leave = new Set();
for (let i=1; i<input.length; i++) {
    let [T, name] = input[i].split(' ');
    T = T.split(':').map(Number);
    if (T[0] < S[0] || (T[0] === S[0] && T[1] <= S[1])) enter.add(name);
    else if (T[0] > E[0] || (T[0] === E[0] && T[1] >= E[1])) {
        if (T[0] < Q[0] || (T[0] === Q[0] && T[1] <= Q[1])) {
            if (enter.has(name)) leave.add(name)
        }
    }
}
console.log(leave.size);
```
