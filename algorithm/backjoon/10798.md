# 세로읽기

```python
li = []
for _ in range(5):
    S = input()
    li.append(S+'*'*14)
res = ''
for j in range(15):
    for i in range(5):
        if li[i][j] != '*':
            res += li[i][j]
print(res)
```

```javascript
const fs = require('fs');
const input = fs.readFileSync('/dev/stdin').toString().split('\n');

let li = [];
for (let i=0; i<5; i++) {
    let S = input[i];
    li.push(S+'***************')
}
let res = '';
for (let j=0; j<15; j++) {
    for (let i=0; i<5; i++) {
        if (li[i][j] !== '*') res += li[i][j];
    }
}
console.log(res);
```
