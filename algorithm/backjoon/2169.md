# 로봇 조종하기

```python
import sys

input = sys.stdin.readline

def solution():
    N, M = map(int, input().split())
    dp = list(map(int, input().split()))
    for i in range(M-1):
        dp[i+1] += dp[i]
    for _ in range(N-2):
        arr = list(map(int, input().split()))
        temp = [0]*M
        temp[0] = arr[0]+dp[0]
        for i in range(1, M):
            temp[i] = arr[i]+max(dp[i], temp[i-1])
        arr[-1] += dp[M-1]
        for i in range(M-2, -1, -1):
            arr[i] += max(dp[i], arr[i+1])
        for i in range(M):
            arr[i] = max(arr[i], temp[i])
        dp = arr
    if N > 1:
        arr = list(map(int, input().split()))
        arr[0] += dp[0]
        for i in range(1, M):
            arr[i] += max(dp[i], arr[i-1])
        dp = arr
    print(dp[-1])

solution()
```
