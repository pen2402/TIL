# 백설 공주와 일곱 난쟁이

```python
import sys

input = sys.stdin.readline

def solution():
    numbers = [False]*100
    total = 0
    for _ in range(9):
        num = int(input())
        numbers[num] = True
        total += num
    for i in range(1, 100):
        if total-100-i >= 100:
            continue
        if numbers[i] and numbers[total-100-i]:
            numbers[i] = numbers[total-100-i] = False
            break
    for i in range(1, 100):
        if numbers[i]:
            print(i)

solution()
```
