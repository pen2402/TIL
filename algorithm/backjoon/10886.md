# 0 = not cute / 1 = cute

```python
import sys

input = sys.stdin.readline

def solution():
    N = int(input())
    total = 0
    for _ in range(N):
        vote = input().rstrip()
        if vote == '1':
            total += 1
    print(f"Junhee is{'' if total > N//2 else ' not'} cute!")

solution()
```
