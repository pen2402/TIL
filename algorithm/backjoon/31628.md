# 가지 한 두름 주세요

```python
import sys

input = sys.stdin.readline

def solution():
    color_set = [[set() for _ in range(10)] for _ in range(2)]
    for i in range(10):
        colors = input().split()
        for j in range(10):
            color_set[0][i].add(colors[j])
            color_set[1][j].add(colors[j])
    print(int(min(min(len(c) for c in color_set[0]), min(len(c) for c in color_set[1])) <= 1))

solution()
```
