# 사분면

```python
import sys

input = sys.stdin.readline

def solution():
    n = int(input())
    res = [['Q1:', 0], ['Q2:', 0], ['Q3:', 0], ['Q4:', 0], ['AXIS:', 0]]
    for _ in range(n):
        x, y = map(int, input().split())
        if x > 0 and y > 0:
            res[0][1] += 1
        elif x > 0 and y < 0:
            res[3][1] += 1
        elif x < 0 and y > 0:
            res[1][1] += 1
        elif x < 0 and y < 0:
            res[2][1] += 1
        else:
            res[4][1] += 1
    for i in range(5):
        print(*res[i])

solution()
```
