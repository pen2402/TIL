import sys

input = sys.stdin.readline

N = int(input())
M = int(input())
material = sorted(map(int, input().split()))
li = [0]*100001
for i in material:
    li[i] += 1

res = 0
for i in material:
    if i >= M-i:
        break
    if M-i <= 100000 and li[M-i] > 0:
        li[M-i] -= 1
        res += 1
print(res)
