# 숨바꼭질 6

```python
N, S = map(int, input().split())
A = list(map(int, input().split()))
A.append(S)
A.sort()
d_list = []
for i in range(len(A)-1):
    d_list.append(A[i+1] - A[i])

d_list.sort()
min_val = d_list.pop(0)
divisors = []
i = 2
while min_val > 1:
    if min_val%i == 0:
        divisors.append(i)
        min_val = min_val//i
    else:
        i += 1

res = 1
for divisor in divisors:
    for d in d_list:
        if d//res % divisor != 0:
            break
    else:
        res *= divisor

print(res)
```
