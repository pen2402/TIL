# 코딩은 예쁘게

```python
def solution():
    N = int(input())
    A = tuple(map(int, input().split()))
    B = tuple(map(int, input().split()))
    res = prev = 0
    for i in range(N):
        d = B[i]-A[i]
        if prev <= 0 and d <= 0:
            d = min(d-prev, 0)
        elif prev >= 0 and d >= 0:
            d = max(d-prev, 0)
        res += abs(d)
        prev = B[i]-A[i]
    print(res)

solution()
```
