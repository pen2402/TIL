# 알고리즘 수업 - 너비 우선 탐색 3

```python
import sys

input = sys.stdin.readline

def solution():
    N, M, R = map(int, input().split())
    graph = [[] for _ in range(N+1)]
    for _ in range(M):
        u, v = map(int, input().split())
        graph[u].append(v)
        graph[v].append(u)
    queue = [(R, 0)]
    visited = [0]*(N+1)
    visited[R] = 1
    while queue:
        idx, d = queue.pop(0)
        for i in graph[idx]:
            if visited[i] == 0:
                visited[i] = visited[idx]+1
                queue.append((i, d+1))
    for i in range(1, N+1):
        print(visited[i]-1)

solution()
```
