# 1, 2, 3, 많다!

```python
import sys

input = sys.stdin.readline

def solution():
    N = int(input())
    for _ in range(N):
        input()
        total = 0
        for c in input()[::4]:
            if c == '!' or total > 9:
                total = 10
                break
            total += int(c)
        print(total if total <= 9 else '!')

solution()
```
