# 셔틀 탈래 말래 탈래 말래 애매하긴 해

```python
import sys

input = sys.stdin.readline

def solution():
    A, B, C, D = map(int, input().split())
    if max(A+B, C) <= D:
        print('~.~')
    elif min(A+B, C) > D:
        print('T.T')
    elif A+B > D >= C:
        print('Walk')
    else:
        print('Shuttle')

solution()
```
