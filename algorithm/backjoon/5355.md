# 화성 수학

```python
import sys

input = sys.stdin.readline

def solution():
    T = int(input())
    for _ in range(T):
        exp = input().split()
        res = float(exp[0])
        for e in exp[1:]:
            if e == '@':
                res *= 3
            elif e == '%':
                res += 5
            else:
                res -= 7
        print(f'{res:.2f}')

solution()
```
