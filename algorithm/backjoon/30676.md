# 이 별은 무슨 색일까

```python
import sys

input = sys.stdin.readline

def solution():
    l = int(input())
    if l >= 620:
        print('Red')
    elif l >= 590:
        print('Orange')
    elif l >= 570:
        print('Yellow')
    elif l >= 495:
        print('Green')
    elif l >= 450:
        print('Blue')
    elif l >= 425:
        print('Indigo')
    else:
        print('Violet')

solution()
```
