# 정육점

```python
import sys

input = sys.stdin.readline

def solution():
    N, M = map(int, input().split())
    meats = []
    total_w = 0
    for _ in range(N):
        w, p = map(int, input().split())
        total_w += w
        meats.append((w, p))
    if total_w < M:
        print(-1)
        return
    meats.sort(key=lambda x: (x[1], -x[0]))
    meats.append((0, 0))
    total_w = total_p = 0
    res = int(1e10)
    for i in range(N):
        w, p = meats[i]
        total_w += w
        total_p += p
        if total_w >= M and total_p < res:
            res = total_p
        if p != meats[i+1][1]:
            total_p = 0
    print(res)

solution()
```
