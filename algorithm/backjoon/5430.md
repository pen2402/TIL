```python
import sys
from collections import deque

input = sys.stdin.readline

T = int(input())
for t in range(T):
    P = list(input().rstrip())
    n = int(input())
    arr = deque(list(input().rstrip()[1:-1].split(',')))
    if not n:
        arr.clear()
    d = 1
    error = False
    for p in P:
        if p == 'R':
            d *= -1
        elif p == 'D':
            if arr:
                if d > 0:
                    arr.popleft()
                else:
                    arr.pop()
            else:
                error = True
                break
    if error:
        print('error')
    elif d > 0:
        print('[' + ','.join(list(arr)) + ']')
    else:
        print('[' + ','.join(list(arr)[::-1]) + ']')
```
