import sys

N = int(sys.stdin.readline())
deck = []
for n in range(N):
    cmd = list(sys.stdin.readline().split())
    if cmd[0] == 'push_front':
        deck = list(reversed(deck))
        deck.append(cmd[1])
        deck = list(reversed(deck))
    if cmd[0] == 'push_back':
        deck.append(cmd[1])
    if cmd[0] == 'pop_front':
        if len(deck) > 0:
            deck = list(reversed(deck))
            print(deck.pop())
            deck = list(reversed(deck))
        else:
            print(-1)
    if cmd[0] == 'pop_back':
        if len(deck) > 0:
            print(deck.pop())
        else:
            print(-1)
    if cmd[0] == 'size':
        print(len(deck))
    if cmd[0] == 'empty':
        print(1) if len(deck) == 0 else print(0)
    if cmd[0] == 'front':
        print(deck[0]) if len(deck) > 0 else print(-1)
    if cmd[0] == 'back':
        print(deck[-1]) if len(deck) > 0 else print(-1)
