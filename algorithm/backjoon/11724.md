# 연결 요소의 개수

```python
import sys

N, M = map(int, sys.stdin.readline().split())
edge = [list(map(int, sys.stdin.readline().split())) for _ in range(M)]
graph = [[0] * N for _ in range(N)]
for u, v in edge:
    graph[u-1][v-1] = 1
    graph[v-1][u-1] = 1

cnt = 0
visited = [0] * N
queue = []
for i in range(N):
    if not visited[i]:
        queue = [i]
        visited[i] = 1
        while queue:
            j = queue.pop(0)
            for k in range(N):
                if graph[j][k] and not visited[k]:
                    queue.append(k)
                    visited[k] = 1
        cnt += 1
print(cnt)
```
