# 지폐 세기

```python
import sys

input = sys.stdin.readline

def solution():
    N = int(input())
    res = 0
    for _ in range(N):
        w = int(input().split()[0])
        if w == 136:
            res += 1
        elif w == 142:
            res += 5
        elif w == 148:
            res += 10
        else:
            res += 50
    print(res*1000)

solution()
```
