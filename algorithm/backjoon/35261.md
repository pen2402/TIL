# 피갤컵 들어왔으면 이 글부터 봐라

```python
import sys

input = sys.stdin.readline

def solution():
    N = int(input())
    S = input()
    res = 5
    for i in range(N-4):
        cnt = 0
        for j in range(5):
            cnt += S[i+j] != 'eagle'[j]
        if cnt < res:
            res = cnt
    print(res)

solution()
```
