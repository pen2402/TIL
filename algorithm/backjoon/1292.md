# 쉽게 푸는 문제

```python
import sys

input = sys.stdin.readline

def solution():
    A, B = map(int, input().split())
    current = i = 1
    res = 0
    while i <= B:
        cnt = 0
        while cnt < current and i <= B:
            cnt += 1
            if i >= A:
                res += current
            i += 1
            if i > B:
                break
        current += 1
    print(res)

solution()
```
