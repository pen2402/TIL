# PyPy3

import sys

input = sys.stdin.readline

N = int(input())
cranes = sorted(map(int, input().split()), reverse=True)
M = int(input())
boxes = sorted(map(int, input().split()), reverse=True)
idx_list = [i for i in range(M)]
res = -1
if boxes[0] <= cranes[0]:
    res = 0
    while idx_list:
        for crane in cranes:
            for idx in idx_list:
                if boxes[idx] <= crane:
                    idx_list.remove(idx)
                    break
        res += 1
print(res)
