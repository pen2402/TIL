# Baseball

```python
import sys

input = sys.stdin.readline

def solution():
    T = int(input())
    for _ in range(T):
        yonsei = korea = 0
        for _ in range(9):
            Y, K = map(int, input().split())
            yonsei += Y
            korea += K
        if yonsei > korea:
            print('Yonsei')
        elif yonsei < korea:
            print('Korea')
        else:
            print('Draw')

solution()
```
