# 와글와글 숭고한

```python
import sys

input = sys.stdin.readline

def solution():
    S, K, H = map(int, input().split())
    min_val = min(S, K, H)
    if S+K+H >= 100:
        print('OK')
    elif S <= min_val:
        print('Soongsil')
    elif K <= min_val:
        print('Korea')
    else:
        print('Hanyang')

solution()
```
