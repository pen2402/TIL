import sys

input = sys.stdin.readline

def solution():
    A = tuple(map(int, input().split()))
    B = tuple(map(int, input().split()))
    scores = [26, 14, 10, 6, 6, 4]
    total_A = 0
    total_B = 3
    for i in range(6):
        total_A += A[i]*scores[i]
        total_B += B[i]*scores[i]
    print('cocjr0208' if total_A > total_B else 'ekwoo')

solution()
