# 골드바흐의 추측

```python
import sys

p = [True] * 1000001
p[0] = p[1] = False

for i in range(2, 1001):
    if p[i]:
        for j in range(i*2, 1000001, i):
            p[j] = False
while 1:
    n = int(sys.stdin.readline())
    if n == 0:
        break
    for i in range(3, n):
        if p[i] and p[n-i]:
            print(f'{n} = {i} + {n-i}')
            break
    else:
        print('Goldbach\'s conjecture is wrong.')
```
