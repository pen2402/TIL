# 애너그램 만들기

```python
import sys

input = sys.stdin.readline

def solution():
    cnt_list = [0]*26
    for c in input().rstrip():
        cnt_list[ord(c)-97] += 1
    for c in input().rstrip():
        cnt_list[ord(c)-97] -= 1
    res = 0
    for cnt in cnt_list:
        res += abs(cnt)
    print(res)

solution()
```
