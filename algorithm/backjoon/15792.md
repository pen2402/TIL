# A/B - 2

```python
import sys

input = sys.stdin.readline

def solution():
    A, B = map(int, input().split())
    res = f'{A//B}.'
    A %= B
    i = 0
    while i < 1000 and A%B > 0:
        if A < B:
            A *= 10
        res += f'{A//B}'
        A %= B
        i += 1
    print(res)

solution()
```
