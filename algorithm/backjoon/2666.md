# 벽장문의 이동

```python
import sys

input = sys.stdin.readline

def solution():
    _ = int(input())
    a, b = map(int, input().split())
    M = int(input())
    if a > b:
        a, b = b, a
    queue = [(a, b, 0)]
    for i in range(M):
        num = int(input())
        _queue = []
        while queue:
            a, b, d = queue.pop()
            if num <= a:
                _queue.append((num, b, d+a-num))
            elif num >= b:
                _queue.append((a, num, d+num-b))
            else:
                _queue.append((a, num, d+b-num))
                _queue.append((num, b, d+num-a))
        queue = _queue
    res = queue[0][2]
    for _, _, d in queue:
        if d < res:
            res = d
    print(res)

solution()
```
