# 카멜레온 부분 문자열

```python
def solution():
    S = input()
    N = len(S)
    lps = [0]*N
    j = 0
    for i in range(1, N):
        while j > 0 and S[i] != S[j]:
            j = lps[j-1]
        if S[i] == S[j]:
            j += 1
            lps[i] = j
    if S[-lps[-1]:] in S[1:-1]:
        print(S[-lps[-1]:])
    elif lps[lps[-1]-1]:
        print(S[-lps[lps[-1]-1]:])
    else:
        print(-1)

solution()
```
