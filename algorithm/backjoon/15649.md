def sequence(N, M):
    if M == 0:
        print(*stack)
        return
    for i in range(1, N+1):
        if not visited[i]:
            stack.append(i)
            visited[i] = 1
            sequence(N, M-1)
            stack.pop()
            visited[i] = 0

N, M = map(int, input().split())
visited = [0] * (N+1)
stack = []
sequence(N, M)
