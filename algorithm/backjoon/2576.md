# 홀수

```python
import sys

input = sys.stdin.readline

def solution():
    total = 0
    min_val = 100
    for _ in range(7):
        n = int(input())
        if n%2:
            total += n
            if n < min_val:
                min_val = n
    if total <= 0:
        print(-1)
        return
    print(total)
    print(min_val)

solution()
```
