# 방 배정하기

```python
import sys

input = sys.stdin.readline

def solution():
    A, B, C, N = map(int, input().split())
    is_full = [False]*(N+1)
    is_full[0] = True
    for i in (A, B, C):
        for j in range(N-i+1):
            is_full[i+j] = is_full[i+j] or is_full[j]
    print(int(is_full[N]))

solution()
```
