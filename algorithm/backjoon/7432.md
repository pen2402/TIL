# 디스크 트리

```python
import sys

input = sys.stdin.readline

class Trie():

    def __init__(self):
        self.root = dict()

    def insert(self, directories):
        node = self.root
        for directory in directories:
            if directory not in node:
                node[directory] = dict()
            node = node[directory]

    def search(self):

        def dfs(node, d):
            for directory in sorted(node.keys()):
                res.append(' '*d+directory)
                dfs(node[directory], d+1)

        res = []
        dfs(self.root, 0)
        return res

def solution():
    N = int(input())
    trie = Trie()
    for _ in range(N):
        directories = input().rstrip().split('\\')
        trie.insert(directories)
    print('\n'.join(trie.search()))

solution()
```
