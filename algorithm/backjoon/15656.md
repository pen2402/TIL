def sequence(N, M):
    if M == 0:
        print(*stack)
        return
    for i in range(N):
        stack.append(A[i])
        sequence(N, M-1)
        stack.pop()

N, M = map(int, input().split())
A = list(map(int, input().split()))
A.sort()
stack = []
