# Previous Level

```python
import sys

input = sys.stdin.readline

def solution():

    def get_level(m):
        m = int(m)
        if m >= 300:
            return 1
        elif m >= 275:
            return 2
        elif m >= 250:
            return 3
        else:
            return 4

    input()
    print(*tuple(map(get_level, input().split())))

solution()
```
