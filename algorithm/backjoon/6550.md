# 부분 문자열

```python
import sys

input = sys.stdin.readlines

for case in input():
    S, T = case.split()
    N = len(S)
    idx = -1
    for i in range(N):
        idx = T.find(S[i], idx+1)
        if idx < 0:
            print('No')
            break
    else:
        print('Yes')
```

```javascript
const fs = require('fs');
const input = fs.readFileSync('/dev/stdin').toString().trim().split('\n');

for (let i=0; i<input.length; i++) {
    let [S, T] = input[i].split(' ');
    let N = S.length;
    let idx = -1;
    for (let j=0; j<N; j++) {
        idx = T.indexOf(S[j], idx+1);
        if (idx < 0) break;
    }
    if (idx < 0) console.log('No');
    else console.log('Yes');
}
```
