# 앞뒤가 맞는 수열

```python
N = int(input())
A = list(map(int, input().split()))[::-1]
Pi = [0]*N
j = 0
for i in range(1, N):
    while j > 0 and A[i] != A[j]:
        j = Pi[j-1]
    if A[i] == A[j]:
        j += 1
        Pi[i] = j
max_len = max(Pi)
if max_len > 0:
    print(max_len, Pi.count(max_len))
else:
    print(-1)
```
