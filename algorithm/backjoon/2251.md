import sys

def fibo(n):
    global cnt_list
    if n >= 2 and len(cnt_list) <= n:
        fibo1 = fibo(n-1)
        fibo2 = fibo(n-2)
        cnt_list.append([fibo1[0]+fibo2[0], fibo1[1]+fibo2[1]])
    return cnt_list[n]

cnt_list = [[1, 0], [0, 1]]

T = int(sys.stdin.readline())
for t in range(T):
    N = int(sys.stdin.readline())
    print(*fibo(N))
