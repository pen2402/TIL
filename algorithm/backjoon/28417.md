# 스케이트보드

```python
import sys

input = sys.stdin.readline

def solution():
    N = int(input())
    res = 0
    for _ in range(N):
        r1, r2, *tricks = map(int, input().split())
        score = max(r1, r2)+sum(sorted(tricks)[-2:])
        if score > res:
            res = score
    print(res)

solution()
```
