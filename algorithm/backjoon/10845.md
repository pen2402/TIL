```python
import sys

N = int(sys.stdin.readline())
q = []
for n in range(N):
    cmd = list(sys.stdin.readline().split())
    if cmd[0] == 'push':
        q.append(cmd[1])
    elif cmd[0] == 'pop':
        if len(q) > 0:
            q = list(reversed(q))
            print(q.pop())
            q = list(reversed(q))
        else:
            print(-1)
    elif cmd[0] == 'size':
        print(len(q))
    elif cmd[0] == 'empty':
        print(0) if len(q) > 0 else print(1)
    elif cmd[0] == 'front':
        print(q[0]) if len(q) > 0 else print(-1)
    elif cmd[0] == 'back':
        print(q[-1]) if len(q) > 0 else print(-1)
```
