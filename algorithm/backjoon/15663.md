def sequence(N, M):
    if M == 0:
        print(*stack)
        return
    for i in range(N):
        if not visited[i] and A[i] not in visited_c[M-1]:
            stack.append(A[i])
            visited[i] = 1
            visited_c[M-1].append(A[i])
            sequence(N, M-1)
            stack.pop()
            visited[i] = 0
    visited_c[M-1] = []

N, M = map(int, input().split())
A = list(map(int, input().split()))
A.sort()
visited = [0] * N
visited_c = [[] for _ in range(M)]
stack = []
sequence(N, M)
