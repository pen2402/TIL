# 집 주소

```python
import sys

input = sys.stdin.readline

def solution():
    while 1:
        N = input().rstrip()
        if N == '0':
            return
        res = 1
        for c in N:
            if c == '0':
                res += 5
            elif c == '1':
                res += 3
            else:
                res += 4
        print(res)

solution()
```
