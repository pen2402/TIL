```python
import sys

input = sys.stdin.readline

def solution():
    R, C = map(int, input().split())
    lines = [input() for _ in range(R)]
    lines.sort()
    res = ['']*9
    D = dict()
    rank = 1
    for line in lines:
        line = line.lstrip('S')
        d = len(line)
        line = line.lstrip('.')
        d -= len(line)
        if line[0] == 'F':
            continue
        idx = int(line[0])-1
        if D.get(d, 0):
            res[idx] = f'{D.get(d, 0)}'
        else:
            res[idx] = D[d] = f'{rank}'
            rank += 1
    print('\n'.join(res))

solution()
```
