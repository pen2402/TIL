import sys

input = sys.stdin.readline

def solution():
    T = int(input())
    res = ''
    for _ in range(T):
        N, K = map(int, input().split())
        if K > 1:
            res += ' '.join(map(str, range(1, N+1)))+'\n'
        elif N in (2, 3):
            res += '-1\n'
        else:
            res += ' '.join(map(str, list(range(2, N+1, 2))+list(range(1, N+1, 2))))+'\n'
    print(res.rstrip())

solution()
