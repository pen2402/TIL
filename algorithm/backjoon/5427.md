import sys

input = sys.stdin.readline

def solution():
    T = int(input())
    delta = [(-1, 0), (0, 1), (1, 0), (0, -1)]
    for _ in range(T):
        w, h = map(int, input().split())
        board = [[*input()] for _ in range(h)]
        stack = []
        fire = []
        for r in range(h):
            for c in range(w):
                if board[r][c] == '@':
                    stack.append((r, c))
                    board[r][c] = '#'
                elif board[r][c] == '*':
                    fire.append((r, c))
                    board[r][c] = '#'
        res = cnt = 0
        while stack:
            cnt += 1
            _fire = []
            while fire:
                r, c = fire.pop()
                for dr, dc in delta:
                    if h > r+dr >= 0 and w > c+dc >= 0:
                        if board[r+dr][c+dc] == '.':
                            _fire.append((r+dr, c+dc))
                            board[r+dr][c+dc] = '#'
            _stack = []
            while stack:
                r, c = stack.pop()
                for dr, dc in delta:
                    if h > r+dr >= 0 and w > c+dc >= 0:
                        if board[r+dr][c+dc] == '.':
                            _stack.append((r+dr, c+dc))
                            board[r+dr][c+dc] = '#'
                    else:
                        res = cnt
                        stack = _stack = []
                        break
            fire = _fire
            stack = _stack
        if res > 0:
            print(res)
        else:
            print('IMPOSSIBLE')

solution()
