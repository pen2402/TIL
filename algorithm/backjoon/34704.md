# 크기가 4인 박스

```python
import sys

input = sys.stdin.readline

def solution():
    input()
    cnt_list = [0]*5
    for a in input()[::2]:
        cnt_list[int(a)] += 1
    res = cnt_list[4]+cnt_list[3]+cnt_list[2]//2+cnt_list[2]%2
    cnt_list[1] -= cnt_list[3]+(cnt_list[2]%2)*2
    if cnt_list[1] > 0:
        res += cnt_list[1]//4+(cnt_list[1]%4 > 0)
    print(res)

solution()
```
