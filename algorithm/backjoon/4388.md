# 받아올림

```python
import sys

input = sys.stdin.readline

def solution():
    while 1:
        a, b = input().split()
        if a == b == '0':
            return
        a = (a[::-1]+'0'*10)[:10]
        b = (b[::-1]+'0'*10)[:10]
        res = carry = 0
        for i in range(10):
            if int(a[i])+int(b[i])+carry >= 10:
                carry = 1
                res += 1
            else:
                carry = 0
        print(res)

solution()
```
