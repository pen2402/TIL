# 링크와 스타트

```python
import sys

def comb(idx, n):
    if idx >= N:
        return
    if n >= 1:
        team_start = 0
        team_link = 0
        for i in range(N):
            for j in range(N):
                if visited[i] and visited[j]:
                    team_start += graph[i][j]
                elif not visited[i] and not visited[j]:
                    team_link += graph[i][j]
        dif = abs(team_start-team_link)
        global res
        if res == -1 or dif < res:
            res = dif
    for i in range(idx, N):
        if not visited[i]:
            visited[i] = 1
            comb(i+1, n+1)
            visited[i] = 0

N = int(sys.stdin.readline())
graph = [list(map(int, sys.stdin.readline().split())) for _ in range(N)]

visited = [0] * N
res = -1
comb(0, 0)
print(res)
```
