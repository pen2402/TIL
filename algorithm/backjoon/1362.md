import sys

input = sys.stdin.readline

def solution():
    t = 1
    while 1:
        o, w = map(int, input().split())
        if o == 0:
            return
        while 1:
            cmd = input().split()
            if cmd[0] == '#':
                break
            elif w <= 0:
                continue
            elif cmd[0] == 'E':
                w -= int(cmd[1])
            else:
                w += int(cmd[1])
        if w <= 0:
            print(f'{t} RIP')
        elif o*2 > w > o//2:
            print(f'{t} :-)')
        else:
            print(f'{t} :-(')
        t += 1

solution()
