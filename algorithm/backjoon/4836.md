```python
import sys

input = sys.stdin.readline

def solution():
    for S in sys.stdin.readlines():
        S = S.rstrip()
        errors = []
        words = S.split()
        N = len(words)
        idx_list = []
        for i in range(N):
            if words[i] == 'dip':
                if i > 0 and words[i-1] == 'jiggle':
                    continue
                elif i > 1 and words[i-2] == 'jiggle':
                    continue
                elif i+1 < N and 'twirl' == words[i+1]:
                    continue
                else:
                    idx_list.append(i)
                    if not errors:
                        errors.append('1')
        if not S.endswith('clap stomp clap'):
            errors.append('2')
        if S.find('twirl') >= 0 > S.find('hop'):
            errors.append('3')
        if S.startswith('jiggle'):
            errors.append('4')
        if S.find('dip') < 0:
            errors.append('5')
        for idx in idx_list:
            words[idx] = 'DIP'
        S = ' '.join(words)
        if len(errors) == 0:
            print('form ok:', S)
        elif len(errors) == 1:
            print(f'form error {errors[0]}:', S)
        else:
            print(f'form errors {", ".join(errors[:-1])} and {errors[-1]}:', S)

solution()
```
