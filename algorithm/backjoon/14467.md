# 소가 길을 건너간 이유 1

```python
import sys

input = sys.stdin.readline

def solution():
    N = int(input())
    cows = [-1]*11
    res = 0
    for _ in range(N):
        num, val = map(int, input().split())
        if cows[num] < 0:
            cows[num] = val
        elif cows[num] != val:
            cows[num] = val
            res += 1
    print(res)

solution()
```
