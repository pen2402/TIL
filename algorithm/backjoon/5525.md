# IOIOI

```python
import sys

N = int(sys.stdin.readline())
M = int(sys.stdin.readline())
S = list(sys.stdin.readline().rstrip())
pattern = 'IO' * N + 'I'
cnt = 0
i = 0
while i <= len(S) - len(pattern):
    for j in range(len(pattern)):
        if S[i+j] != pattern[j]:
            i += 1
            break
    else:
        cnt += 1
        i += 2
print(cnt)
```
