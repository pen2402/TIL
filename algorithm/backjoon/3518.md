# 공백왕 빈-칸

```python
import sys

input = sys.stdin.readline

def solution():
    docs = sys.stdin.readlines()
    length_list = [0]*90
    for doc in docs:
        words = doc.split()
        for i in range(len(words)):
            length = len(words[i])
            if length > length_list[i]:
                length_list[i] = length
    for doc in docs:
        words = doc.split()
        res = ''
        for i in range(len(words)):
            length = len(words[i])
            res += words[i]+' '*(length_list[i]-length+1)
        print(res.rstrip())

solution()
```
