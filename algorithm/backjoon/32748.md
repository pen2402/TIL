import sys

input = sys.stdin.readline

def solution():
    f = input()[::2]
    total = 0
    for S in input().split():
        total += int(''.join(str(f.find(c)) for c in S))
    res = ''
    while total > 0:
        res = str(f[total%10])+res
        total //= 10
    print(res)

solution()
