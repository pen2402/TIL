# 후문으로

```python
import sys

input = sys.stdin.readline

def solution():
    N = int(input())
    res = 1
    for _ in range(N):
        A, B = map(int, input().split())
        d = abs(A-B)
        if A*B < 0:
            d -= 1
        if A > B:
            res -= d
        else:
            res += d
    if res <= 0:
        res -= 1
    print(res)

solution()
```
