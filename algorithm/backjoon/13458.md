# 시험 감독

```python
import sys

input = sys.stdin.readline

def solution():
    N = int(input())
    A = tuple(map(int, input().split()))
    B, C = map(int, input().split())
    res = N
    for a in A:
        if B >= a:
            continue
        a -= B
        res += a//C+(a%C > 0)
    print(res)

solution()
```
