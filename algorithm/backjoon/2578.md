```python
my_bingo = []
for i in range(5):
    my_bingo.extend(map(int, input().split()))

call = []
for i in range(5):
    call.extend(map(int, input().split()))

turn = 0
for num in call:
    turn += 1
    idx = my_bingo.index(num)
    my_bingo[idx] = 0
    cnt = 0
    for i in range(5):
        if any(my_bingo[i*5:(i+1)*5]) == False:
            cnt += 1
        if any(my_bingo[i::5]) == False:
            cnt += 1
    if any(my_bingo[0::6]) == False:
        cnt += 1
    if any(my_bingo[4:21:4]) == False:
        cnt += 1
    if cnt >= 3:
        break
print(turn)
```
