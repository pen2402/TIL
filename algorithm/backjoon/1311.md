# 할 일 정하기 1

```python
# PyPy3

import sys

input = sys.stdin.readline
INF = int(1e6)

def dfs(idx, visited):
    if visited == (1 << N)-1:
        return 0
    if dp[idx][visited]:
        return dp[idx][visited]
    dp[idx][visited] = INF
    for i in range(N):
        if visited & (1 << i):
            continue
        next_val = dfs(idx+1, visited | (1 << i))
        dp[idx][visited] = min(dp[idx][visited], D[idx][i] + next_val)
    return dp[idx][visited]

N = int(input())
D = [list(map(int, input().split())) for _ in range(N)]
dp = [[0]*(1 << 21) for _ in range(N)]
print(dfs(0, 0))
```
