# 수빈이와 수열

```python
import sys

input = sys.stdin.readline

def solution():
    N = int(input())
    B = tuple(map(int, input().split()))
    total = 0
    res = ''
    for i in range(N):
        a = B[i]*(i+1)-total
        total += a
        res += f'{a} '
    print(res.rstrip())

solution()
```
