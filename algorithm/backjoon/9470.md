```python
import sys

input = sys.stdin.readline

T = int(input())
for t in range(T):
    K, M, P = map(int, input().split())
    tree = [[] for _ in range(M+1)]
    tree_r = [[] for _ in range(M+1)]
    D = [0] * (M+1)
    for _ in range(P):
        a, b = map(int, input().split())
        tree[a].append(b)
        tree_r[b].append(a)
        D[b] += 1

    queue = []
    strahler = [0] * (M+1)
    for i in range(1, M+1):
        if not D[i]:
            queue.append(i)
            strahler[i] = 1

    while queue:
        idx = queue.pop(0)
        for i in tree[idx]:
            D[i] -= 1
            if not D[i]:
                queue.append(i)
                
                cnt = 0
                max_val = 0
                for j in tree_r[i]:
                    if strahler[j] > max_val:
                        cnt = 1
                        max_val = strahler[j]
                    elif strahler[j] == max_val:
                        cnt += 1

                strahler[i] = max_val
                if cnt >= 2:
                    strahler[i] += 1

    print(K, max(strahler))
```
