import sys

input = sys.stdin.readline

L = int(input())
S = [0]+sorted(map(int, input().split()))+[1001]
N = int(input())
idx = -1
res = 0
for i in range(L+1):
    if N >= S[i] and N <= S[i+1]:
        idx = i
        break
for i in range(S[idx]+1, S[idx+1]):
    for j in range(i+1, S[idx+1]):
        if N >= i and N <= j:
            res += 1
print(res)
