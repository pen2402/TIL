# 월간 향유회 시즌 종료

```python
import sys

input = sys.stdin.readline

def solution():
    N, K = map(int, input().split())
    max_scores = [0]*K
    numbers = [-1]*(K+1)
    for i in range(N):
        scores = tuple(map(int, input().split()))
        for j in range(K):
            if scores[j] > max_scores[j]:
                max_scores[j] = scores[j]
                numbers[j] = i
            elif scores[j] == max_scores[j]:
                numbers[j] = -1
    print(len(set(numbers))-1)

solution()
```
