# 문자열 로또

```python
import sys

input = sys.stdin.readline

def solution():
    L, N = map(int, input().split())
    lotto = [input() for _ in range(N)]
    scores = {}
    K = int(input())
    for i in range(N):
        for j in range(L-K+1):
            s = lotto[i][j:j+K]
            scores[s] = scores.get(s, 0)+1
    print(max(scores.values()))

solution()
```
