# 아 마이마이 하고 싶다

```python
import sys
from heapq import heappush, heappop

input = sys.stdin.readline

def solution():
    N, M, K = map(int, input().split())
    heap = [0]*M
    for t in map(int, input().split()):
        heappush(heap, t+heappop(heap))
    print('GO' if heappop(heap) > K else 'WAIT')

solution()
```
