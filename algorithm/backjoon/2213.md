# 트리의 독립집합

```python
import sys

sys.setrecursionlimit(10002)
input = sys.stdin.readline

def dfs(idx):
    dp[idx][1] = w[idx]
    S = [set(), set()]
    for i in graph[idx]:
        if not visited[i]:
            visited[i] = 1
            temp = dfs(i)
            dp[idx][0] += max(dp[i])
            dp[idx][1] += dp[i][0]
            if dp[i][0] > dp[i][1]:
                S[0] = S[0].union(temp[0])
            else:
                S[0] = S[0].union(temp[1])
            S[1] = S[1].union(temp[0])
    return S[0], S[1]|{idx}

n = int(input())
w = [0]+list(map(int, input().split()))
graph = [[] for _ in range(n+1)]
dp = [[0]*2 for _ in range(n+1)]
visited = [0]*(n+1)
for _ in range(n-1):
    a, b = map(int, input().split())
    graph[a].append(b)
    graph[b].append(a)
visited[1] = 1
res = dfs(1)
if dp[1][0] > dp[1][1]:
    print(dp[1][0])
    print(*res[0])
else:
    print(dp[1][1])
    print(*res[1])
```
