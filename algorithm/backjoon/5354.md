# J박스

```python
import sys

input = sys.stdin.readline

def solution():
    T = int(input())
    for t in range(T):
        N = int(input())
        print('#'*N)
        for _ in range(N-2):
            print('#'+'J'*(N-2)+'#')
        if N > 1:
            print('#'*N)
        if t < T-1:
            print()

solution()
```
