# 조합 0의 개수

```python
import sys

n, m = map(int, sys.stdin.readline().split())
a = n-m
val = 1
while n > a:
    val *= n
    n -= 1
while m > 1:
    val /= m
    m -= 1
while a > 1:
    val /= a
    a -= 1

S = list(str(int(val)))
cnt = 0
for i in range(len(S)):
    if S.pop() == '0':
        cnt +=1
    else:
        break
print(cnt)
```
