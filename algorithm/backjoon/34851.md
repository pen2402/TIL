# 사칙연산 게임

```python
import sys

input = sys.stdin.readline

def solution():
    N = input()
    res = i = 0
    for a in map(int, input().split()):
        if a == 1 or i == 0 or i == res == 1:
            res += a
        else:
            res *= a
        res %= 1000000007
        i += 1
    print(res)

solution()
```
