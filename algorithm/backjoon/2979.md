# 트럭 주차

```python
import sys

input = sys.stdin.readline

def solution():
    A, B, C = map(int, input().split())
    times = [0]*101
    for _ in range(3):
        a, b = map(int, input().split())
        times[a:b] = [t+1 for t in times[a:b]]
    res = 0
    for t in times:
        if t == 1:
            res += A
        elif t == 2:
            res += B*2
        elif t == 3:
            res += C*3
    print(res)

solution()
```
