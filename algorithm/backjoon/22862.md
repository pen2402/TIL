# 가장 긴 짝수 연속한 부분 수열 (large)

```python
def solution():
    N, K = map(int, input().split())
    S = list(map(int, input().split()))
    arr = [0]
    for i in range(N):
        if S[i]%2:
            arr.append(arr[-1])
        else:
            arr[-1] += 1
    res = arr[:K+1][-1]
    M = len(arr)
    for i in range(K+1, M):
        if arr[i]-arr[i-(K+1)] > res:
            res = arr[i]-arr[i-(K+1)]
    print(res)

solution()
```
