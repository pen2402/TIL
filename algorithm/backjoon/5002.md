# 도어맨

```python
import sys

input = sys.stdin.readline

def solution():
    X = int(input())
    S = [*input().rstrip()]
    N = len(S)
    S += '.'
    m = w = 0
    for i in range(N):
        if S[i] == 'M':
            if m-w < X:
                m += 1
            elif S[i+1] == 'W':
                w += 1
                S[i+1] = 'M'
            else:
                print(i)
                return
        else:
            if w-m < X:
                w += 1
            elif S[i+1] == 'M':
                m += 1
            else:
                print(i)
                return
    else:
        print(N)

solution()
```
