def star(idx):
    global stars
    if idx >= N:
        return
    new_stars = []
    for i in range(len(stars)):
        new_stars.append(stars[i] + [' '] + stars[i])
        stars[i] = ['   ' * (idx//3)] + stars[i] + ['   ' * (idx//3)]
    stars += new_stars
    star(idx*2)

N = int(input())
stars = [['  *  '], [' * * '], ['*****']]
star(3)
for i in range(len(stars)):
    print(*stars[i], sep='')
