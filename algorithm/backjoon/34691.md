# 대전과학고등학교를 사랑하십니까?

```python
import sys

input = sys.stdin.readline

def solution():
    while 1:
        c = input()[0]
        if c == 'a':
            print('Panthera tigris')
        elif c == 't':
            print('Pinus densiflora')
        elif c == 'f':
            print('Forsythia koreana')
        else:
            return

solution()
```
