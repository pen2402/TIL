# 선물할인

```python
import sys

input = sys.stdin.readline

def solution():
    n, b, a = map(int, input().split())
    gifts = sorted(map(lambda x: int(x)//2, input().split()))
    total = res = i = 0
    for j in range(n):
        total += gifts[j]
        if j >= a:
            total += gifts[i]
            i += 1
        if total > b:
            break
        if j+1 > res:
            res = j+1
    print(res)

solution()
```
