def sequence(N, M):
    if M == 0:
        print(*stack)
        return
    for i in range(1, N+1):
        if not stack or (stack and i >= stack[-1]):
            stack.append(i)
            sequence(N, M-1)
            stack.pop()

N, M = map(int, input().split())
stack = []
sequence(N, M)
