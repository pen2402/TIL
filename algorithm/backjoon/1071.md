def solution():
    N = int(input())
    res = sorted(map(int, input().split()))
    cnt = 0
    for i in range(N-1):
        if i > 0 and res[i] == res[i-1]:
            cnt += 1
        else:
            cnt = 1
        if res[i]+1 != res[i+1]:
            continue
        if res[i+1] == res[-1]:
            print(*res[:i-cnt+1]+res[i-cnt+1:][::-1])
            return
        for j in range(i+2, N):
            if res[j] > res[i+1]:
                res[i+1], res[j] = res[j], res[i+1]
                break
    print(*res)

solution()
