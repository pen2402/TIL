# 먼 카드

```python
import sys

input = sys.stdin.readline

def solution():
    N = int(input())
    starts = [-1]*(N+1)
    res = i = 0
    for x in map(int, input().split()):
        if starts[x] < 0:
            starts[x] = i
        elif i-starts[x]-1 > res:
            res = i-starts[x]-1
        i += 1
    print(res)

solution()
```
