# 경기 결과

```python
import sys

input = sys.stdin.readline

def solution():
    N = int(input())
    win_a = win_b = 0
    for _ in range(N):
        A, B = map(int, input().split())
        if A == B:
            continue
        elif A > B:
            win_a += 1
        else:
            win_b += 1
    print(win_a, win_b)

solution()
```
