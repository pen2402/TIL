```python
import sys

input = sys.stdin.readline

N = int(input())
line = [[*map(int, input().split())]]
for _ in range(N-1):
    x, y = map(int, input().split())
    for i in range(len(line)):
        if line[i][0] >= x >= line[i][1] or line[i][0] <= y <= line[i][1]:
            if x < line[i][0]:
                line[i][0] = x
            elif y > line[i][1]:
                line[i][1] = y
            break
    else:
        line.append([x, y])

line.sort()
start = line[0][0]
end = line[0][1]
res = 0
for i in range(1, len(line)):
    if line[i][0] > end:
        res += end-start
        start = line[i][0]
        end = line[i][1]
    elif line[i][1] > end:
        end = line[i][1]
res += end-start
print(res)
```
