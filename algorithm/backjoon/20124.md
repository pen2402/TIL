# 모르고리즘 회장님 추천 받습니다

```python
import sys

input = sys.stdin.readline

def solution():
    N = int(input())
    res = ''
    max_score = 0
    for _ in range(N):
        name, score = input().split()
        score = int(score)
        if score > max_score:
            max_score = score
            res = name
        elif score == max_score and name < res:
            res = name
    print(res)

solution()
```
