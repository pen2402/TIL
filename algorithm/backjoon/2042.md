# 구간 합 구하기

```python
import sys

input = sys.stdin.readline

def update(idx):
    tree[idx] = tree[idx*2]+tree[idx*2+1]
    if idx <= 1:
        return
    update(idx//2)

def sum_up(l, r):
    res = 0
    while l < r:
        if l%2:
            res += tree[l]
            l += 1
        if r%2 == 0:
            res += tree[r]
            r -= 1
        l, r = l//2, r//2
    if l == r:
        res += tree[l]
    return res

N, M, K = map(int, input().split())
n = 1
while n < N:
    n *= 2
tree = [0]*(n*2)
tree[n:n+N] = [int(input()) for _ in range(N)]
for i in range(n-1, 0, -1):
    tree[i] = tree[i*2]+tree[i*2+1]
for _ in range(M+K):
    a, b, c = map(int, input().split())
    if a == 1:
        tree[n+b-1] = c
        update((n+b-1)//2)
    else:
        print(sum_up(b+n-1, c+n-1))
```
