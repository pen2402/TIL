# Fly me to the Alpha Centauri

```python
def d(n):
    a = 1
    b = 1
    count = 0
    while(1):
        for j in range(2):
            for k in range(a):
                if count >= n:
                    return print(b)
                count += 1
            if count >= n:
                return print(b)
            b += 1
        if count >= n:
            return print(b)
        a += 1
    return print(b)

T = int(input())
for t in range(T):
    x,y = map(int, input().split())
    d(y-x)
```
