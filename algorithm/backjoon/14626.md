# ISBN

```python
import sys

input = sys.stdin.readline

def solution():
    ISBN = input()
    value = 0
    idx = 0
    for i in range(12):
        if ISBN[i] == '*':
            idx = i
        else:
            value += int(ISBN[i])*((i%2)*2+1)
    r = (20-int(ISBN[12])-value%10)%10
    if idx%2:
        if r%3 == 2:
            r += 10
        elif r%3 == 1:
            r += 20
        print(r//3)
    else:
        print(r)

solution()
```
