# 수열의 변화

```python
import sys

input = sys.stdin.readline

def solution():
    N, K = map(int, input().split())
    arr = list(map(int, input().split(',')))
    for i in range(1, K+1):
        _arr = [0]*(N-i)
        for j in range(N-i):
            _arr[j] = arr[j+1]-arr[j]
        arr = _arr
    print(','.join(map(str, arr)))

solution()
```
