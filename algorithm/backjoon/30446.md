import sys

input = sys.stdin.readline

def solution():
    n = int(input())
    res = 0
    for i in range(1, 100000):
        s = str(i)
        number = int(s[:-1]+s[::-1])
        if number > n:
            break
        res += 1
        number = int(s+s[::-1])
        if number <= n:
            res += 1
    print(res)

solution()
