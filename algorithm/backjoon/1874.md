# 스택 수열

```python
import sys
N = int(sys.stdin.readline())
num_list = []
for n in range(N):
    num_list += [int(sys.stdin.readline())]

stack = [i for i in range(1, num_list[0]+1)]
pushpop = ['+'] * num_list[0]
max_val = num_list[0]
for j in range(len(num_list)):
    if len(stack) != 0 and stack[-1] == num_list[j]:
        stack = stack[:-1]
        pushpop += ['-']
    elif max_val < num_list[j]:
        stack += [i for i in range(max_val+1, num_list[j])]
        pushpop += ['+'] * (num_list[j] - max_val) + ['-']
        max_val = num_list[j]
    else:
        print('NO')
        break
else:
    print('\n'.join(pushpop))
```
