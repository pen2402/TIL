def stars(n):
    global star
    if n >= N:
        star[1] = '*'
        if n == 1:
            star = ['*']
        return
    star[0] = '* ' + star[0] + '**'
    for i in range(1, len(star)):
        star[i] = '* ' + star[i] + ' *'
    star = ['*'*(len(star)+2)] + ['*' + ' '*(len(star)+1)] + star
    star = star + ['*' + ' '*(len(star)-2) + '*'] + ['*'*len(star)]
    return stars(n+1)

N = int(input())
star = ['*', '*', '*']
stars(1)
print('\n'.join(star))
