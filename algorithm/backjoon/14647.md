# 준오는 조류혐오야!!

```python
import sys

input = sys.stdin.readline

def solution():
    n, m = map(int, input().split())
    total = max_cnt = 0
    cnt_list = [0]*m
    for _ in range(n):
        s = input()
        cnt = s.count('9')
        total += cnt
        if cnt > max_cnt:
            max_cnt = cnt
        s = s.split()
        for i in range(m):
            cnt_list[i] += s[i].count('9')
    print(total-max(max_cnt, max(cnt_list)))

solution()
```
