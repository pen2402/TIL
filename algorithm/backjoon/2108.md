# 통계학

```python
import sys
N = int(sys.stdin.readline())
l = []
cnt = [0] * 8001
for i in range(N):
    n = int(sys.stdin.readline())
    l.append(n)
    cnt[n+4000] += 1
l.sort()

print('%.f'%(sum(l)/N))
print(l[N//2])
if cnt.count(max(cnt)) > 1:
    print(cnt.index(max(cnt),cnt.index(max(cnt))+1)-4000)
else:
    print(cnt.index(max(cnt))-4000)
print(max(l)-min(l))
```
