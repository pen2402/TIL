# 정수 삼각형

```python
import sys

N = int(sys.stdin.readline())
triangle = []
for i in range(N):
    triangle.append(list(map(int, sys.stdin.readline().split())))

max_sum = 0
for n in range(2**(N-1)):
    route = bin(n)[2:].zfill(N-1)
    i = j = 0
    sum_number = triangle[0][0]
    for r in route:
        i += 1
        j += int(r)
        sum_number += triangle[i][j]
    if sum_number > max_sum:
        max_sum = sum_number

print(max_sum)
```
