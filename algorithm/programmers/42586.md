# 기능개발

```python
def solution(progresses, speeds):
    N = len(speeds)  # 작업 개수
    queue = []  # 소요 기간 큐
    res = []  # 결과(배포일별 배포 기능 수) 저장 리스트
    for i in range(N):
        temp = 100-progresses[i]  # 남은 작업량
        days = temp//speeds[i] + bool(temp%speeds[i])*1  # 남은 작업량을 작업 속도로 나눈 몫 + 나머지 올림한 수
        queue.append(days)  # 소요 기간 큐에 추가
    
    last = 0  # 이전 작업의 배포일
    while queue:
        day = queue.pop(0)  # 차례대로 소요 기간을 pop
        if last < day:  # 이전 작업 배포일보다 늦으면
            res.append(1)  # 배포일 추가 후 이전 작업의 배포일 갱신
            last = day
        else:  # 이전 작업 배포일보다 빠르면
            res[-1] += 1  # 이전 배포일의 기능 수 증가(last의 초깃값은 0, 소요 기간은 최소 1이므로 에러 발생하지 않음)
    return res
```
