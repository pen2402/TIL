# 언어별 개발자 분류하기

```mysql
WITH FRONT AS (
    SELECT SUM(CODE) AS SUM_CODE
    FROM SKILLCODES
    WHERE CATEGORY = 'Front End'
),
PYTHON AS (
    SELECT CODE
    FROM SKILLCODES
    WHERE NAME = 'Python'
),
C_SHARP AS (
    SELECT CODE
    FROM SKILLCODES
    WHERE NAME = 'C#'
),
DEVELOPERS_GRADES AS (
    SELECT
        CASE
            WHEN
                SKILL_CODE&(SELECT SUM_CODE FROM FRONT) > 0
                AND SKILL_CODE&(SELECT CODE FROM PYTHON) > 0 THEN 'A'
            WHEN SKILL_CODE&(SELECT CODE FROM C_SHARP) > 0 THEN 'B'
            WHEN SKILL_CODE&(SELECT SUM_CODE FROM FRONT) > 0 THEN 'C'
            ELSE NULL
        END AS GRADE,
        ID,
        EMAIL
    FROM DEVELOPERS
)

SELECT *
FROM DEVELOPERS_GRADES
WHERE GRADE IS NOT NULL
ORDER BY GRADE, ID;
```
