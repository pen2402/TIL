# 저자 별 카테고리 별 매출액 집계하기

```mysql
WITH SALES_SUMS AS (
    SELECT
        BOOK_ID,
        SUM(SALES) AS TOTAL_SALES
    FROM BOOK_SALES
    WHERE SALES_DATE BETWEEN '2022-01-01' AND '2022-01-31'
    GROUP BY BOOK_ID
)

SELECT
    A.AUTHOR_ID,
    A.AUTHOR_NAME,
    B.CATEGORY,
    SUM(B.PRICE*C.TOTAL_SALES) AS TOTAL_SALES
FROM
    AUTHOR A
    JOIN BOOK B ON A.AUTHOR_ID = B.AUTHOR_ID
    LEFT JOIN SALES_SUMS C ON B.BOOK_ID = C.BOOK_ID
GROUP BY A.AUTHOR_ID, B.CATEGORY
ORDER BY A.AUTHOR_ID, B.CATEGORY DESC;
```
