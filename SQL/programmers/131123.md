# 즐겨찾기가 가장 많은 식당 정보 출력하기

```mysql
SELECT
    A.FOOD_TYPE,
    A.REST_ID,
    A.REST_NAME,
    A.FAVORITES
FROM
    REST_INFO A
    JOIN (
        SELECT
            FOOD_TYPE,
            MAX(FAVORITES) AS MAX_FAVORITES
        FROM REST_INFO
        GROUP BY FOOD_TYPE
    ) B
    ON A.FOOD_TYPE = B.FOOD_TYPE
    AND A.FAVORITES = B.MAX_FAVORITES
ORDER BY A.FOOD_TYPE DESC;
```
